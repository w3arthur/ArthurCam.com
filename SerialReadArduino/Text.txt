void serialEvent()
{
    while (!Serial.available());
    String action = Serial.readString();
    action.trim();
    Serial.println(action);
    if (action.equals("true") == 0 || action.equals("1") == 0 )
        digitalWrite(3, !digitalRead(3));
  delay(1);
}
